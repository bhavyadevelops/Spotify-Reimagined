#include <iostream>
#include <vector>
#include <stack>
#include <algorithm>
using namespace std;

// Student structure
struct Student 
{
    int id;
    string name;
    int marks;
    string course;
};

vector<Student> list1;     // stores students
stack<Student> deleted;    // for undo delete

// Add student
void addStudent() 
{
    Student s;
    cout << "Enter ID: ";
    cin >> s.id;
    cout << "Enter Name: ";
    cin >> s.name;
    cout << "Enter Marks: ";
    cin >> s.marks;
    cout << "Enter Course: ";
    cin >> s.course;

    list1.push_back(s);
    cout << "Student added successfully!\n";
}

// Display students
void displayStudents() 
{
    if (list1.empty()) 
    {
        cout << "No records found.\n";
        return;
    }

    cout << "\nStudent Records:\n";
    for (auto s : list1) 
    {
        cout << "ID: " << s.id
             << " | Name: " << s.name
             << " | Marks: " << s.marks
             << " | Course: " << s.course << endl;
    }
}

// Delete student
void deleteStudent(int id) 
{
    for (int i = 0; i < list1.size(); i++) 
    {
        if (list1[i].id == id) 
        {
            deleted.push(list1[i]); // store for undo
            list1.erase(list1.begin() + i);
            cout << "Student deleted!\n";
            return;
        }
    }
    cout << "Student not found.\n";
}

// Undo delete
void undoDelete() 
{
    if (!deleted.empty()) 
    {
        list1.push_back(deleted.top());
        deleted.pop();
        cout << "Undo successful!\n";
    } 
    else 
    {
        cout << "Nothing to undo.\n";
    }
}

// Search student
void searchStudent(int id) 
{
    for (auto s : list1) 
    {
        if (s.id == id) 
        {
            cout << "Student Found:\n";
            cout << "ID: " << s.id
                 << " | Name: " << s.name
                 << " | Marks: " << s.marks
                 << " | Course: " << s.course << endl;
            return;
        }
    }
    cout << "Student not found.\n";
}

// Sort students by marks (descending)
void sortStudents() 
{
    sort(list1.begin(), list1.end(), [](Student a, Student b) 
    {
        return a.marks > b.marks;
    });

    cout << "Students sorted by marks (highest first).\n";
}

// Main menu
int main() 
{
    int choice, id;

    do 
    {
        cout << "\n===== MENU =====\n";
        cout << "1. Add Student\n";
        cout << "2. Delete Student\n";
        cout << "3. Undo Delete\n";
        cout << "4. Search Student\n";
        cout << "5. Display Students\n";
        cout << "6. Sort Students\n";
        cout << "7. Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        switch (choice) 
        {
            case 1:
                addStudent();
                break;
            case 2:
                cout << "Enter ID to delete: ";
                cin >> id;
                deleteStudent(id);
                break;
            case 3:
                undoDelete();
                break;
            case 4:
                cout << "Enter ID to search: ";
                cin >> id;
                searchStudent(id);
                break;
            case 5:
                displayStudents();
                break;
            case 6:
                sortStudents();
                break;
            case 7:
                cout << "Exiting...\n";
                break;
            default:
                cout << "Invalid choice!\n";
        }

    } while (choice != 7);

    return 0;
}
